import os
import re
import sys
import requests

def download_cloud_mail_video(url, output_filename="downloaded_video.mp4"):
    """
    Cloud Mail (Mail.ru Cloud) paylaşılan URL'sinden videoyu tespit eder ve indirir.
    """
    headers = {
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36",
        "Accept-Language": "tr-TR,tr;q=0.9,en-US;q=0.8,en;q=0.7",
    }

    print("[*] Sayfa kaynak kodu analiz ediliyor...")
    
    try:
        session = requests.Session()
        response = session.get(url, headers=headers, timeout=15)
        response.raise_for_status()
    except Exception as e:
        print(f"[-] Sayfaya erişilemedi: {e}")
        return

    # Mail.ru video konfigürasyonunu içeren JSON/text yapısını regex ile yakalıyoruz
    # "video" ve "video_sources" genellikle inline script tagleri içinde nesne olarak bulunur
    html_content = response.text
    
    # 1. Doğrudan video indirme URL'sini yakalamaya çalış (En yüksek kalite .mp4)
    video_urls = re.findall(r'"url"\s*:\s*"([^"]+?\.mp4[^"]*?)"', html_content)
    
    if not video_urls:
        # Alternatif regex: Genel video kaynak nesnelerini ara
        video_urls = re.findall(r'https://cloclo[^"]+?\.mp4[^"]*', html_content)

    if not video_urls:
        print("[-] Sayfa içinde video kaynak URL'si tespit edilemedi.")
        print("[!] URL'nin herkese açık (public) olduğundan ve şifre korumalı olmadığından emin ol.")
        return

    # Unicode kaçış karakterlerini temizle (\u002F -> /)
    download_url = video_urls[0].encode().decode('unicode-escape')
    
    # Eğer URL her ihtimale karşı temizlenmemişse regex kalıntılarını temizle
    download_url = download_url.replace('\\/', '/')

    print(f"[+] Video URL'si başarıyla tespit edildi.")
    print(f"[*] İndirme başlıyor: {output_filename}")

    # Stream modunda indirmeyi başlat (Büyük dosyalar için RAM'i şişirmez)
    try:
        video_response = session.get(download_url, headers=headers, stream=True, timeout=30)
        video_response.raise_for_status()
        
        total_size = int(video_response.headers.get('content-length', 0))
        downloaded = 0

        with open(output_filename, 'wb') as f:
            for chunk in video_response.iter_content(chunk_size=1024 * 1024): # 1 MB'lık parçalar
                if chunk:
                    f.write(chunk)
                    downloaded += len(chunk)
                    if total_size:
                        percent = (downloaded / total_size) * 100
                        sys.stdout.write(f"\r Progress: %{percent:.2f} [{downloaded // (1024*1024)}MB / {total_size // (1024*1024)}MB]")
                        sys.stdout.flush()
                    else:
                        sys.stdout.write(f"\r İndirilen: {downloaded // (1024*1024)} MB")
                        sys.stdout.flush()
                        
        print(f"\n[+] İndirme tamamlandı! Dosya: {os.path.abspath(output_filename)}")

    except Exception as e:
        print(f"\n[-] Video indirilirken hata oluştu: {e}")

if __name__ == "__main__":
    # Test etmek istediğin Cloud Mail URL'sini buraya yapıştır
    cloud_url = "https://cloud.mail.ru/public/XXXX/YYYYYYYYY"
    
    # Çıktı dosya adı
    output_file = "cloud_video.mp4"
    
    if "cloud.mail.ru" not in cloud_url:
        print("[-] Lütfen geçerli bir cloud.mail.ru linki girin.")
    else:
        download_cloud_mail_video(cloud_url, output_file)
